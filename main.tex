\section{Fomular for Scientists}
Maxwell Equations
\begin{align*}
    \div{\vb{D}} &= 0 \\
    \div{\vb{B}} &= 0 \\
    \curl{\vb{E}} &= -\pdv{\vb{B}}{t} \\
    \curl{\vb{H}} &= \pdv{\vb{D}}{t}
\end{align*}
Schroedinger Equation
$$
    i\hbar \pdv{\ket{\varphi}}{t} = \mathcal{H} \ket{\varphi}
$$
